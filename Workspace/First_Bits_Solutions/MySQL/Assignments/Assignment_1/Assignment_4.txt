Q2.
Ans. SELECT DEPARTMENT_ID FROM EMPLOYEES;

Q3.
Ans. SELECT * FROM EMPLOYEES
     ORDER BY FIRST_NAME DESC;

Q4.
Ans. select FIRST_NAME,LAST_NAME,SALARY,SALARY *0.15 AS PF FROM EMPLOYEES;

Q5.
Ans. SELECT EMPLOYEE_ID,FIRST_NAME,LAST_NAME,SALARY FROM EMPLOYEES
     ORDER BY SALARY ASC;

Q6.
Ans. SELECT SUM(SALARY) AS TOTAL_SALARY_PALABLE FROM EMPLOYEES;

Q7.
Ans. SELECT MAX(SALARY) AS HIGHEST_SALARY,
     MIN(SALARY) AS LOWEST_SALARY FROM EMPLOYEES;

Q8.
Ans. SELECT AVG(SALARY) AS AVERAGE_SALARY, 
     COUNT(EMPLOYEE_ID) AS 
     TOTAL_NUMBER_OF_EMPLOYEE FROM EMPLOYEES;

Q9.
Ans. SELECT COUNT(EMPLOYEE_ID) FROM EMPLOYEES;

Q10.
Ans. SELECT COUNT(DISTINCT JOB_ID) FROM EMPLOYEES;

Q11.
Ans. SELECT * FROM EMPLOYEES LIMIT 10;

Q12.
Ans. SELECT * FROM EMPLOYEES
     WHERE SALARY BETWEEN 10000 AND 15000;

Q13.
Ans. SELECT FIRST_NAME ,LAST_NAME,DEPARTMENT_ID FROM EMPLOYEES

Q14.
Ans.  SELECT FIRST_NAME,LAST_NAME,SALARY FROM EMPLOYEES
      WHERE DEPARTMENT_ID IN (30,100) AND SALARY NOT BETWEEN 10000 AND 15000;

Q15.
